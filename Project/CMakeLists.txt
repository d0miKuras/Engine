cmake_minimum_required(VERSION 3.7)
project(Pong)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/")
find_package(SDL2 REQUIRED)
find_package(SDL2_image REQUIRED)
if(SDL2_image)
	message("Package found")
else()
	message("Package not found")
endif()

file(GLOB ENGINE_HEADERS "src/engine/*.h")
file(GLOB ENGINE_SOURCES "src/engine/*.cpp")
file(GLOB GAME_HEADERS "src/game/*.h")
file(GLOB GAME_SOURCES "src/game/*.cpp")
add_subdirectory(libs/csv-parser)
include_directories(${SDL2_INCLUDE_DIRS} ${SDL2_IMAGE_INCLUDE_DIRS} libs/csv-parser/include src)
message("=================================================" + ${SDL2_IMAGE_INCLUDE_DIRS})
add_executable(Pong ${ENGINE_HEADERS} ${ENGINE_SOURCES} ${GAME_HEADERS} ${GAME_SOURCES})
target_link_libraries(Pong ${SDL2_LIBRARIES} ${SDL2_IMAGE_LIBRARIES} csv)
