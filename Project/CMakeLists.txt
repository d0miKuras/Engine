cmake_minimum_required(VERSION 3.7)
project(Pong)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/")
find_package(SDL2 REQUIRED)
find_package(SDL2_image REQUIRED)
if(SDL2_image)
	message("Package found")
else()
	message("Package not found")
endif()

add_subdirectory(libs/csv-parser)
include_directories(${SDL2_INCLUDE_DIRS} ${SDL2_IMAGE_INCLUDE_DIRS} libs/csv-parser/include)
message("=================================================" + ${SDL2_IMAGE_INCLUDE_DIRS})
add_executable(Pong GameProgrammingOverview.cpp Game.cpp Actor.cpp Component.cpp SpriteComponent.cpp AnimSpriteComponent.cpp BGSpriteComponent.cpp TileMapComponent.cpp MovementComponent.cpp Ship.cpp Math.cpp Random.cpp Game.h Math.h SpriteComponent.h AnimSpriteComponent.h BGSpriteComponent.h TileMapComponent.h MovementComponent.h Component.h Actor.h Ship.h Random.h)
target_link_libraries(Pong ${SDL2_LIBRARIES} ${SDL2_IMAGE_LIBRARIES} csv)
